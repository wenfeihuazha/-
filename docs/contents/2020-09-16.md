# 关于微任务宏任务加上async简单练习

## 问题
```js
async function async1 (){
  console.log('A');
  async2();
  console.log('B')
};
async function async2 (){
  console.log('C');
};
console.log('D');
setTimeout(function(){
  console.log('F')
},0);
async1();
new Promise(function(resolve){
  console.log('G');
  resolve();
}).then(function(){
  console.log('H')
});
console.log('I');
```
